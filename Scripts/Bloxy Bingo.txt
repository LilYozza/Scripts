local SubContainer = Player.PlayerGui:WaitForChild("Bingo").StaticDisplayArea.Cards.PlayerArea.Cards.Container.SubContainer

-->Non Return Functions<--
function firebutton(button)
    if button ~= nil then
       for i,signal in pairs(getconnections(button.MouseButton1Click)) do
           signal:Fire()
       end
       for i,signal in pairs(getconnections(button.MouseButton1Down)) do
           signal:Fire()
       end
       for i,signal in pairs(getconnections(button.Activated)) do
           signal:Fire()
       end
    end
end

Auto:Toggle("Auto Mark", function(bool)
    shared.ToggleAM = bool
    task.spawn(function()
        while shared.ToggleAM do
            pcall(function()
                local Cards = SubContainer:FindFirstChild("Blocks") and SubContainer.Blocks.Block
                    or SubContainer.VerticalScroll.Cards

                local BingoButton = SubContainer.Buttons.ClaimButton

                if Cards and BingoButton then
                    for _, card in pairs(Cards:GetChildren()) do
                        if card:IsA("Frame") then
                            local numbers = card:FindFirstChild("Content")
                                and card.Content:FindFirstChild("Numbers")

                            if numbers then
                                for _, button in pairs(numbers:GetChildren()) do
                                    task.delay(0, function()
                                        firebutton(button)
                                    end)
                                end

                                local toGoText = card:FindFirstChild("ToGo")
                                    and card.ToGo:FindFirstChild("ToGoText")

                                if toGoText and toGoText.Text == "BINGO!" then
                                    firebutton(BingoButton)
                                end
                            end
                        end
                    end
                end
            end)
            task.wait(0.1)
        end
    end)
end)
