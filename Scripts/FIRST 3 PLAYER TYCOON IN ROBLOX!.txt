local library = loadstring(game:HttpGet(('https://raw.githubusercontent.com/LilYozza/Scripts/main/Misc/UILibrary.txt')))()w = library:CreateWindow("Main")

local Players=game:GetService("Players")
local Player=game.Players.LocalPlayer

local Auto = w:CreateFolder("Auto")
local Item = w:CreateFolder("Items")

local Tycoon, TCollect, ItemSelect
local PName = {}
local Tycoons = workspace:FindFirstChild("Tycoons")
local ProtectionBlocks = workspace:FindFirstChild("ProtectionBlocks")
local suffixes = {"K", "M", "B", "T", "Q"}
local Swords = {"Final Hero", "Darkheart", "ClassicSword", "StarterSword", "SCAR"}
local Items = {}


local function Touch(p,t)firetouchinterest(Player.Character:WaitForChild("HumanoidRootPart"),p,0)firetouchinterest(Player.Character:WaitForChild("HumanoidRootPart"),p,1)task.wait(t)end
local function fromSuffixString(s) local suffixes = {"K", "M", "B", "T", "Q"} found = false if s == "0" then return 0 end if s:match("$") then s = string.gsub(s, '%$', "") end if s:match(",") then s = string.gsub(s, '%,', "") end if s:match("+") then s = string.gsub(s, '%+', "") end for i, value in ipairs(suffixes) do if s:match(value) then found = true end end if found == false then return tonumber(s) end local n, suffix = string.match(s, "(.*)(%a)$") if n and suffix then local i = table.find(suffixes, suffix) or 0 return tonumber(n) * math.pow(10, i * 3) end return tonumber(s) end
local function UpdateItems()
    if Items ~= nil then table.clear(Items) end
    for _, v in pairs(Tycoons:GetDescendants()) do
        if v:IsA("TouchTransmitter") and v:FindFirstAncestor("Giver") and not v:FindFirstAncestor("Essentials") and not table.find(Items, v.Parent.Parent.Name) then
            table.insert(Items, v.Parent.Parent.Name)
        end
    end
    if Items ~= nil then table.sort(Items, function(a, b) return a:byte() < b:byte() end) end
end UpdateItems()
local function GetTycoon()
    for key, value in pairs(Tycoons:GetDescendants()) do
        if value.Name == "Owner" and value.Value == Player then
            Tycoon = value.Parent
            PName = Tycoon.Name:split(" ")
            PName = string.gsub(Tycoon.Name, PName[#PName], "").."worker"
            break
        end
    end
end
local function ClaimTycoon()
    if Tycoon == nil then
        for key, value in pairs(Tycoon:GetDescendants()) do
            if value.Name == "Owner" and value.Value == nil then
                for i, part in pairs(value.Parent:FindFirstChild("Entrance")) do
                    if part.Name == "Head" then
                        Touch(Part)
                        Tycoon = value.Parent
                        break
                    end
                end
            end
        end
    end
end

Auto:Toggle("Auto Click",function(bool)
    shared.toggle1 = bool
    spawn (function()
        if Tycoon == nil then GetTycoon() end
        while shared.toggle1 do
            for key, value in pairs(Tycoons:FindFirstChild(PName):GetDescendants()) do
                if value:isA("ClickDetector") and value:FindFirstAncestor("Mine") and shared.toggle1 and value:FindFirstAncestor("Essentials").Parent.Name:match(Tycoon.Name:split(" ")[1]) then
                    fireclickdetector(value)task.wait()break
                end    
            end
            wait()
        end
    end)
end)

Auto:Toggle("Auto Collect",function(bool)
    shared.toggle2 = bool
    spawn (function()
        if Tycoon == nil then GetTycoon() end
        while shared.toggle2 do
            for key, value in pairs(Tycoon:GetDescendants()) do
                if value.Name == "Giver" and value:FindFirstAncestor("Essentials") and shared.toggle2 then
                    Touch(value)task.wait()
                end
            end
            task.wait(1)
        end
    end)
end)
Auto:Toggle("Smart Collect",function(bool)
    shared.toggle5 = bool
    TCollect = shared.toggle5
end)

Auto:Toggle("Auto Buy",function(bool)
    shared.toggle3 = bool
    spawn (function()
        if Tycoon == nil then GetTycoon() end
        while shared.toggle3 do
            for _, v in pairs(Tycoon:FindFirstChild("Buttons"):GetDescendants()) do
                if v:IsA("TouchTransmitter") and v.Parent.Transparency == 0 then
                    if TCollect and v.Parent.Parent:FindFirstChild("Price").Value <= fromSuffixString(Player.leaderstats.Blocks.Value) + Tycoon:FindFirstChild("CurrencyToCollect").Value then
                        Touch(Tycoon:FindFirstChild("Essentials"):FindFirstChild("Giver"),1)
                    end
                    if v.Parent.Parent:FindFirstChild("Price").Value <= fromSuffixString(Player.leaderstats.Blocks.Value) then
                        Touch(v.Parent,1)
                    end
                end
            end
        	task.wait()
        end
    end)
end)

Auto:Toggle("Auto Heal",function(bool)
    shared.toggle4 = bool
    spawn (function()
        while shared.toggle4 do
            if Player.character:WaitForChild("Humanoid").Health <= 50 then
                for key, value in pairs(Tycoons:GetDescendants()) do
                    if value.Name == "Giver" and value:FindFirstAncestor("MedkitGiver") then
                        Touch(value)
                        repeat wait() until Player.Backpack:WaitForChild("Medkit") ~= nil
                        Medkit = Player.Backpack:WaitForChild("Medkit")
                        Player.character:WaitForChild("Humanoid"):EquipTool(Medkit)
                        virtual:ClickButton1(Vector2.new(0, 0), workspace.CurrentCamera.CFrame)
                        for i, part in pairs(Swords) do
                            if Player.Backpack:FindFirstChild(part) then
                                weapon = Player.Backpack:FindFirstChild(part)
                                Player.character:WaitForChild("Humanoid"):EquipTool(weapon)
                                break
                            end
                        end
                        break
                    end
                end
            end
        	wait()
        end
    end)
end)

Auto:Button("Remove Doors",function()
    if ProtectionBlocks then ProtectionBlocks:Destroy() end
    for key, value in pairs(Tycoons:GetDescendants()) do
        if value.Name == "OwnerOnlyDoor" then
            value:Destroy()
        end
    end
end)
