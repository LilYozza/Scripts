local suffixes = {"K", "M", "B", "T", "Q"}
local function fromSuffixString(s)
    found = false
    if s == "0" then return 0 end
    if s:match("$") then s = string.gsub(s, '%$', "") end
    if s:match(",") then s = string.gsub(s, '%,', "") end
    if s:match("+") then s = string.gsub(s, '%+', "") end
    for i, value in ipairs(suffixes) do
        if s:match(value) then found = true end
    end
    if found == false then return tonumber(s) end
	local n, suffix = string.match(s, "(.*)(%a)$")
	if n and suffix then
		local i = table.find(suffixes, suffix) or 0
		return tonumber(n) * math.pow(10, i * 3)
	end
	return tonumber(s)
end
function GetTycoon()
    for _,v in pairs(Tycoons:GetDescendants()) do
        if v:IsA("ObjectValue") and v.Value == Player then
            Tycoon = v.Parent
        end
    end
end GetTycoon()

Auto:Toggle("Auto Collect",function(bool)
    shared.toggle1 = bool
    spawn (function()
        if not Tycoon then GetTycoon() end
        while shared.toggle1 do
            for _,v in pairs(Tycoon:FindFirstChild("Essentials"):GetDescendants()) do
                if v:IsA("TouchTransmitter") and v:FindFirstAncestor("Giver") and shared.toggle1 then
                    firetouchinterest(Player.character:WaitForChild("HumanoidRootPart"), v.Parent, 0)
                    firetouchinterest(Player.character:WaitForChild("HumanoidRootPart"), v.Parent, 1) wait(1) break
                end
            end
        	wait()
        end
    end)
end)
Auto:Toggle("Smart Collect",function(bool)
    shared.toggle2 = bool
    SCollect = shared.toggle2
end)

Auto:Toggle("Auto Buy",function(bool)
    shared.toggle3 = bool
    spawn (function()
        if not Tycoon then GetTycoon() end
        while shared.toggle3 do
            for _,v in pairs(Tycoon:FindFirstChild("Buttons"):GetDescendants()) do
                if v:IsA("TouchTransmitter") and v.Parent.Transparency == 0 and shared.toggle3 and not v.Parent.Parent:FindFirstChild("Gamepass") then
                    if SCollect and v.Parent.Parent:FindFirstChild("Price").Value <= fromSuffixString(Player.leaderstats.Cash.Value) + Tycoon:FindFirstChild("CurrencyToCollect").Value then
                        firetouchinterest(Player.character:WaitForChild("HumanoidRootPart"), Tycoon:FindFirstChild("Essentials"):FindFirstChild("Giver"), 0)
                        firetouchinterest(Player.character:WaitForChild("HumanoidRootPart"), Tycoon:FindFirstChild("Essentials"):FindFirstChild("Giver"), 1) wait(1)
                    end
                    if v.Parent.Parent:FindFirstChild("Price").Value <= fromSuffixString(Player.leaderstats.Cash.Value) then
                        firetouchinterest(Player.character:WaitForChild("HumanoidRootPart"), v.Parent, 0)
                        firetouchinterest(Player.character:WaitForChild("HumanoidRootPart"), v.Parent, 1) wait(1)
                    end
                    
                end
            end
        	wait()
        end
    end)
end)
