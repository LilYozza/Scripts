local Interactables = workspace:FindFirstChild("Interactables")
local Coins = Interactables:FindFirstChild("Coins")

local QuestItems = Interactables:FindFirstChild("QuestItems")

local w = library:CreateWindow("Main")
local Auto = w:CreateFolder("Auto")
local Misc = w:CreateFolder("Misc")

Auto:Toggle("Auto Coins",function(bool)
    shared.toggle1 = bool
    spawn (function()
        while shared.toggle1 do
            Player = game.Players.LocalPlayer
            pcall(function()
                for key, value in pairs(Coins:GetChildren()) do
                    if value.Name ~= "ReplaceCoin" then
                        firetouchinterest(Player.character:WaitForChild("HumanoidRootPart"), value, 0)
                        firetouchinterest(Player.character:WaitForChild("HumanoidRootPart"), value, 1)
                        wait()
                    end
                end
                wait()
            end)
        end
    end)
end)

Auto:Toggle("Auto Artifact",function(bool)
    shared.toggle2 = bool
    spawn (function()
        while shared.toggle2 do
            for key, value in pairs(Artifacts:GetChildren()) do
                if not shared.toggle2 then break end 
                back = Player.character:WaitForChild("HumanoidRootPart").CFrame
                Player.character:WaitForChild("HumanoidRootPart").CFrame = value.CFrame
                task.wait(1)
                keypress(0x45)
                task.wait(2)
                keyrelease(0x45)
                Player.character:WaitForChild("HumanoidRootPart").CFrame = back
            end
            wait()
        end
    end)
end)

Auto:Toggle("Auto QuestItem",function(bool)
    shared.toggle3 = bool
    spawn (function()
        while shared.toggle3 do
            for key, value in pairs(QuestItems:GetChildren()) do
                back = Player.character:WaitForChild("HumanoidRootPart").CFrame
                Player.character:WaitForChild("HumanoidRootPart").CFrame = value.CFrame
                wait(1)
                keypress(0x45)
                wait(2)
                keyrelease(0x45)
                Player.character:WaitForChild("HumanoidRootPart").CFrame = back
            end
            wait()
        end
    end)
end)

Auto:Toggle("Auto Sell",function(bool)
    shared.toggle4 = bool
    spawn (function()
        while shared.toggle4 do
            if game.Players.LocalPlayer.GoldCoins.value > 0 then
                local Event = game:GetService("ReplicatedStorage").SellAll
                Event:FireServer()
                wait(0.5)
            end
            wait()
        end
    end)
end)

Auto:Toggle("Auto Heal",function(bool)
    shared.toggle5 = bool
    spawn (function()
        while shared.toggle5 do
            if Player.character:WaitForChild("Humanoid").Health < 100 then
                Event = game:GetService("ReplicatedStorage").HealHumanoid
                Event:FireServer(Player.character:WaitForChild("Humanoid"))
                wait()
               end
            wait()
        end
    end)
end)

Auto:Toggle("Auto Oxygen",function(bool)
    shared.toggle6 = bool
    spawn (function()
        while shared.toggle6 do
            Player = game.Players.LocalPlayer
            if Player.ScubaGear.AirTank then
                if Player.ScubaGear.AirTank.Value < 270 then
                    local Event = game:GetService("ReplicatedStorage").ChangeValue
                    Event:FireServer(Player.ScubaGear.AirTank, 270)
                    wait()
                end
            end
            if Player.ScubaGear.SuperScuba then
                if Player.ScubaGear.SuperScuba.Value < 270 then
                    local Event = game:GetService("ReplicatedStorage").ChangeValue
                    Event:FireServer(Player.ScubaGear.SuperScuba, 270)
                    wait()
                end
            end
            wait()
        end
    end)
end)
