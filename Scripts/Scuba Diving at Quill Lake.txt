local library = loadstring(game:HttpGet(('https://raw.githubusercontent.com/bloodball/-back-ups-for-libs/main/wall%20v3')))()
local FullBright = loadstring(game:HttpGet('https://raw.githubusercontent.com/LilYozza/Scripts/main/Misc/FullBright.txt'))

local Player = game.Players.LocalPlayer
local character = Player.Character or Player.CharacterAdded:Wait()
local root = character:WaitForChild("HumanoidRootPart")
local hum = character:WaitForChild("Humanoid")
local virtual = game:GetService("VirtualUser")

local Interactables = workspace:FindFirstChild("Interactables")
local Coins = Interactables:FindFirstChild("Coins")
local Artifacts = Interactables:FindFirstChild("Artifacts")
local QuestItems = Interactables:FindFirstChild("QuestItems")

local w = library:CreateWindow("Main")
local Auto = w:CreateFolder("Auto")
local Misc = w:CreateFolder("Misc")

Auto:Toggle("Auto Coins",function(bool)
    shared.toggle1 = bool
    spawn (function()
        while shared.toggle1 do
            Player = game.Players.LocalPlayer
            pcall(function()
                for key, value in pairs(Coins:GetChildren()) do
                    if value.Name ~= "ReplaceCoin" then
                        firetouchinterest(Player.character:WaitForChild("HumanoidRootPart"), value, 0)
                        firetouchinterest(Player.character:WaitForChild("HumanoidRootPart"), value, 1)
                        wait()
                    end
                end
                wait()
            end)
        end
    end)
end)

Auto:Toggle("Auto Artifact",function(bool)
    shared.toggle2 = bool
    spawn (function()
        while shared.toggle2 do
            pcall(function()
                Player = game.Players.LocalPlayer
                for key, value in pairs(Artifacts:GetChildren()) do
                local back = root.CFrame
                root.CFrame = part.CFrame
                wait(1)
                keypress(0x45)
                wait(2)
                keyrelease(0x45)
                root.CFrame = back
                end
                wait()
            end)
        end
    end)
end)

Auto:Toggle("Auto Sell",function(bool)
    shared.toggle4 = bool
    spawn (function()
        while shared.toggle4 do
            pcall(function()
                Player = game.Players.LocalPlayer
                if Player.GoldCoins.value > 0 then
                    local Event = game:GetService("ReplicatedStorage").SellAll
                    Event:FireServer()
                    wait(0.5)
                end
                wait()
            end)
        end
    end)
end)

Auto:Toggle("Auto Heal",function(bool)
    shared.toggle5 = bool
    spawn (function()
        while shared.toggle5 do
            pcall(function()
                Player = game.Players.LocalPlayer
                if Player.character:WaitForChild("Humanoid").Health < 100 then
                    local Event = game:GetService("ReplicatedStorage").HealHumanoid
                    Event:FireServer(Player.character:WaitForChild("Humanoid"))
                    wait()
                end
                wait()
            end)
        end
    end)
end)

Misc:Toggle("Anti AFK",function(bool)
    shared.Misc1 = bool
    spawn (function()
        if shared.Misc1 then
            game:GetService("Players").LocalPlayer.Idled:connect(function()
            virtual:Button2Down(Vector2.new(0,0),workspace.CurrentCamera.CFrame)
            wait(1)
            virtual:Button2Up(Vector2.new(0,0),workspace.CurrentCamera.CFrame)
            end)
        end
    end)
end)

Misc:Toggle("Full Bright",function(bool)
    shared.Misc2 = bool
    spawn (function()
        FullBright()
    end)
end)

Misc:Toggle("Infinite Jump",function(bool)
    shared.Misc3 = bool
    spawn (function()
        game:GetService("UserInputService").JumpRequest:Connect(function()
            if shared.Misc3 then
                Player.character:WaitForChild("Humanoid"):ChangeState("Jumping")
            end
        end)
    end)
end)

Misc:Toggle("Teleport",function(bool)
    shared.Misc4 = bool
    spawn(function()
        local mouse = Player:GetMouse()
        mouse.KeyDown:Connect(function(key)
            if key == "b" and shared.Misc4 then
                if mouse.Target then
                    Player.character:WaitForChild("HumanoidRootPart").CFrame = CFrame.new(mouse.Hit.x, mouse.Hit.y + 5, mouse.Hit.z)
                end
            end
        end)
    end)
end)

local Speed = Misc:Slider("Slider",{
    ['default'] = 16;
    ['min'] = 0;
    ['max'] = 100;
    ['precise'] = true;
},function(value)
    game:GetService("Players").LocalPlayer.Character.Humanoid.WalkSpeed = value
    local Players = game:GetService("Players")
end)

Misc:Button("3rd Person",function()
    game:GetService('Players').LocalPlayer.CameraMode = 'Classic'
    Player.CameraMaxZoomDistance = 150
    Player.CameraMinZoomDistance = 0
end)

Misc:DestroyGui()
