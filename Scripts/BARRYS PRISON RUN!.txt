
local Players=game:GetService("Players")
local Player=game.Players.LocalPlayer

local Auto = w:CreateFolder("Auto")

local LocalObjects = Workspace:FindFirstChild("LocalObjects")
local GameObstacles = Workspace:FindFirstChild("GameObstacles")
local Spawns = LocalObjects:WaitForChild("spawns")

local function Touch(p,t)firetouchinterest(Player.Character:WaitForChild("HumanoidRootPart"),p,0)firetouchinterest(Player.Character:WaitForChild("HumanoidRootPart"),p,1)task.wait(t)end

Auto:Toggle("Auto Complete",function(bool)
    shared.toggle1 = bool
    spawn (function()
        while shared.toggle1 do
            for _ = Player.leaderstats.Stage.Value+1, #Spawns:getChildren(), 1 do
                if not shared.toggle1 then break end
                Touch(Spawns:WaitForChild("p"..Player.leaderstats.Stage.Value+1):WaitForChild("go"),1)
                _ = Player.leaderstats.Stage.Value+1
            end
        	task.wait()
        end
    end)
end)

Auto:Button("Give Items (WIP)",function()
    spawn(function()
        for _, v in pairs(workspace:GetDescendants()) do
            if v.Name:match("Giver") then
                for key, value in pairs(v:GetDescendants()) do
                    if value:isA("TouchTransmitter") then
                        print("Giving",v)
                        firetouchinterest(Player.character:WaitForChild("HumanoidRootPart"), value.Parent, 0)
                        firetouchinterest(Player.character:WaitForChild("HumanoidRootPart"), value.Parent, 1)
                    end
                end
            end
        end
    end)
end)

Auto:Button("Remove Obstacles",function()
    if GameObstacles then GameObstacles:Destroy() end
end)
