local Player=game.Players.LocalPlayer

local Auto = w:CreateFolder("Auto")

local Tycoons = workspace:FindFirstChild("Tycoons")

local function fromSuffixString(s) local suffixes = {"K", "M", "B", "T", "Q"} found = false if s == "0" then return 0 end if s:match("$") then s = string.gsub(s, '%$', "") end if s:match(",") then s = string.gsub(s, '%,', "") end if s:match("+") then s = string.gsub(s, '%+', "") end for i, value in ipairs(suffixes) do if s:match(value) then found = true end end if found == false then return tonumber(s) end local n, suffix = string.match(s, "(.*)(%a)$") if n and suffix then local i = table.find(suffixes, suffix) or 0 return tonumber(n) * math.pow(10, i * 3) end return tonumber(s) end
local function Touch(p,t)firetouchinterest(Player.Character:WaitForChild("HumanoidRootPart"),p,0)firetouchinterest(Player.Character:WaitForChild("HumanoidRootPart"),p,1)task.wait(t)end

for _, v in pairs(Tycoons:GetDescendants()) do
    if v:IsA("StringValue") and v.Value == Player.Name then
        Tycoon = v.Parent
    end
end

Auto:Toggle("Auto Buy",function(bool)
    shared.toggle1 = bool
    spawn (function()
        while shared.toggle1 do
            for _, v in pairs(Tycoon:FindFirstChild("Buttons"):GetDescendants()) do
                if v:isA("TouchTransmitter") then
                    price = v.Parent.Parent:FindFirstChild("InfoPart"):FindFirstChild("Info"):FindFirstChild("Price").Text
                    money = Player.PlayerGui.Main.Top.MoneyDisplay.Amount.Text
                    if not price:match("ROBUX") and fromSuffixString(price) <= fromSuffixString(money) and shared.toggle1 then
                    Touch(v.Parent,0.7)break
                    end
                end
            end
        	task.wait()
        end
    end)
end)

Auto:Toggle("Auto Upgrade",function(bool)
    shared.toggle2 = bool
    spawn (function()
        while shared.toggle2 do
            for _, v in pairs(Tycoon:FindFirstChild("Tycoon"):GetDescendants()) do
                if v:isA("Humanoid") and tonumber(v.Parent:FindFirstChild("Info"):FindFirstChild("PenguinInfo"):FindFirstChild("PenguinLevel").text:split(" ")[2]:split("/")[1]) < 10  then
                    price = tonumber(fromSuffixString(Player.PlayerGui.Main.Center.Penguins.List:FindFirstChild(v.Parent.Name).PenguinPrice.Text))
                    money = tonumber(fromSuffixString(Player.PlayerGui.Main.Top.MoneyDisplay.Amount.Text))
                    if price <= money then
                        Event = game:GetService("ReplicatedStorage").Remotes.Customization
                        Event:InvokeServer("Upgrade", v.Parent)
                        task.wait()
                    end
                end
            end
        	wait()
        end
    end)
end)

Auto:Toggle("Auto Fish",function(bool)
    shared.toggle3 = bool
    spawn (function()
        while shared.toggle3 do
            Event = game:GetService("ReplicatedStorage").Remotes.ReelFish
            Event:InvokeServer(Vector3.new(0, 0, 0), true)
        	task.wait(1)
        end
    end)
end)

Auto:Button("Spawn Teleport",function()

end)
