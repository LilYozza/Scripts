local function fromSuffixString(s)
    found = false
    if s:match("$") then s = string.gsub(s, '%$', "") end
    if s:match(",") then s = string.gsub(s, '%,', "") end
    if s:match("+") then s = string.gsub(s, '%+', "") end
    if s:upper():match("ROBUX") or s:upper():match("R") then return s end
    for i, value in ipairs(suffixes) do
        if s:match(value) then found = true end
    end
    if found == false then return tonumber(s) end
	local n, suffix = string.match(s, "(.*)(%a)$")
	if n and suffix then
		local i = table.find(suffixes, suffix) or 0
		return tonumber(n) * math.pow(10, i * 3)
	end
	return tonumber(s)
end


Auto:Toggle("Auto Buy",function(bool)
    shared.toggle1 = bool
    spawn (function()
        while shared.toggle1 do
            for _, v in pairs(Tycoon:FindFirstChild("Buttons"):GetDescendants()) do
                if v:isA("TouchTransmitter") then
                    price = v.Parent.Parent:FindFirstChild("InfoPart"):FindFirstChild("Info"):FindFirstChild("Price").Text
                    money = Player.PlayerGui.Main.Top.MoneyDisplay.Amount.Text
                    if not price:match("ROBUX") and fromSuffixString(price) <= fromSuffixString(money) and shared.toggle1 then
                        firetouchinterest(Player.character:WaitForChild("HumanoidRootPart"), v.Parent, 0)
                        firetouchinterest(Player.character:WaitForChild("HumanoidRootPart"), v.Parent, 1)
                        wait(0.7)
			break
                    end
                end
            end
        	wait()
        end
    end)
end)

Auto:Toggle("Auto Upgrade",function(bool)
    shared.toggle2 = bool
    spawn (function()
        while shared.toggle2 do
            for _, v in pairs(Tycoon:FindFirstChild("Tycoon"):GetDescendants()) do
                if v:isA("Humanoid") and tonumber(v.Parent:FindFirstChild("Info"):FindFirstChild("PenguinInfo"):FindFirstChild("PenguinLevel").text:split(" ")[2]:split("/")[1]) < 10  then
                    price = tonumber(fromSuffixString(Player.PlayerGui.Main.Center.Penguins.List:FindFirstChild(v.Parent.Name).PenguinPrice.Text))
                    money = tonumber(fromSuffixString(Player.PlayerGui.Main.Top.MoneyDisplay.Amount.Text))
                    if price <= money then
                        Event = game:GetService("ReplicatedStorage").Remotes.Customization
                        Event:InvokeServer("Upgrade", v.Parent)
                        wait()
                    end
                end
            end
        	wait()
        end
    end)
end)
