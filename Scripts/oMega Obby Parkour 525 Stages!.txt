local Player=game.Players.LocalPlayer

local Auto = w:CreateFolder("Auto")
local Stages = workspace:FindFirstChild("Stages")
local rebirth = workspace:FindFirstChild("Play Now!")

local function Touch(p,t)firetouchinterest(Player.Character:WaitForChild("HumanoidRootPart"),p,0)firetouchinterest(Player.Character:WaitForChild("HumanoidRootPart"),p,1)task.wait(t)end

local function Rebirth()
    for _, v in pairs(rebirth:GetDescendants()) do
        if v:IsA("TouchTransmitter") then
            Touch(v.Parent)break
        end
    end
end

Auto:Toggle("Auto Complete",function(bool)
    shared.toggle1 = bool
    spawn (function()
        for _=Player.leaderstats.Stage.Value+1, #Stages:GetChildren(), 1 do
            if not shared.toggle1 then break end
            Touch(Stages:FindFirstChild(_):FindFirstChild("Spawn"),0.5)
        end
        if shared.toggle2 and Player.leaderstats.Stage.Value >= 524 then Rebirth() end
    end)
end)

Auto:Toggle("Auto Rebirth",function(bool)
    shared.toggle2 = bool
end)
