local library = loadstring(game:HttpGet(('https://raw.githubusercontent.com/LilYozza/Scripts/main/Misc/UILibrary.txt')))()w = library:CreateWindow("Main")

local Players=game:GetService("Players")
local Player=game.Players.LocalPlayer

local Auto = w:CreateFolder("Auto")
local Other = w:CreateFolder("Other")

local RCart, Cart
local PathGivers = workspace:FindFirstChild("PathGivers")

for _,v in pairs(workspace:GetDescendants()) do
    if v:IsA("ClickDetector") and v.Parent.Parent.Parent.Name:match("Carts") and not Cart then
        v.Parent.Parent.Parent.Name = "Carts1" Cart =  v.Parent.Parent.Parent
    end
    if v:IsA("ClickDetector") and v.Parent.Name:match("Regen") and not RCart then
        v.Parent.Parent.Name = "Carts2" RCart =  v.Parent.Parent
    end
end
Auto:Toggle("Speed Up",function(bool)
    shared.toggle1 = bool
    spawn (function()
        while shared.toggle1 do
            for _,v in pairs(Cart:GetDescendants()) do
                if v:IsA("ClickDetector") and v.Parent.Name:match("Up") and shared.toggle1 then
                    fireclickdetector(v) wait()
                end
            end
        	wait()
        end
    end)
end)

Auto:Toggle("Speed Down",function(bool)
    shared.toggle2 = bool
    spawn (function()
        while shared.toggle2 do
            for _,v in pairs(Cart:GetDescendants()) do
                if v:IsA("ClickDetector") and v.Parent.Name:match("Down") and shared.toggle2 then
                    fireclickdetector(v) wait()
                end
            end
        	wait()
        end
    end)
end)

Auto:Button("Break",function()
    spawn(function()
        for _,v in pairs(Cart:GetDescendants()) do
            if v:IsA("ClickDetector") and v.Parent.Name:match("On") then
                fireclickdetector(v) wait()
            end
        end
    end)
end)

Auto:Button("Spawn",function()
    spawn(function()
        for _,v in pairs(RCart:GetDescendants()) do
            if v:IsA("ClickDetector") and v.Parent.Name:match("Regen") then
                fireclickdetector(v) wait()
            end
        end
    end)
end)

Other:Button("Give Items",function()
    spawn(function()
        for _,v in pairs(PathGivers:GetDescendants()) do
            if v:IsA("TouchTransmitter") then
                firetouchinterest(Player.character:WaitForChild("HumanoidRootPart"), v.Parent, 0)
                firetouchinterest(Player.character:WaitForChild("HumanoidRootPart"), v.Parent, 0) wait()
            end
        end
    end)
end)

Other:Button("Teleport End",function()
    spawn(function()
        for _,v in pairs(workspace:GetDescendants()) do
            if v:IsA("TouchTransmitter") and v.Parent.Name == "SpawnLocation" then
                Player.character:WaitForChild("HumanoidRootPart").CFrame = v.Parent.CFrame
            end
        end
    end)
end)
