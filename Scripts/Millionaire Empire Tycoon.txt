local library = loadstring(game:HttpGet(('https://raw.githubusercontent.com/LilYozza/Scripts/main/Misc/UILibrary.txt')))()
local FullBright = loadstring(game:HttpGet('https://raw.githubusercontent.com/LilYozza/Scripts/main/Misc/FullBright.txt'))
local ESP = loadstring(game:HttpGet('https://raw.githubusercontent.com/LilYozza/Scripts/main/Misc/ESP.txt'))
local NameTag = loadstring(game:HttpGet('https://raw.githubusercontent.com/LilYozza/Scripts/main/Misc/NameTags.txt'))

local Player = game.Players.LocalPlayer
local character = Player.Character or Player.CharacterAdded:Wait()
local root = character:WaitForChild("HumanoidRootPart")

local virtual = game:GetService("VirtualUser")

local Tycoon, SCollect, Collection, TycoonItems
local TycoonKit = workspace:FindFirstChild("Tycoon Kit")
local Tycoons = TycoonKit:FindFirstChild("Tycoons")

local w = library:CreateWindow("Main")
local Auto = w:CreateFolder("Auto")
local Misc = w:CreateFolder("Misc")

function GetTycoon()
    for _,v in pairs(Tycoons:GetDescendants()) do
        if v:IsA("StringValue") and v.Value == Player.Name then
            Tycoon = v.Parent.Parent
        end
    end
    TycoonItems = Tycoon:FindFirstChild("TycoonItems")
    for _,v in pairs(TycoonItems:FindFirstChild("Essentials"):GetDescendants()) do if v:IsA("TouchTransmitter") and v:FindFirstAncestor("Giver") then Collection = v.Parent end end
end GetTycoon()

Auto:Toggle("Auto Collect",function(bool)
    shared.toggle1 = bool
    spawn (function()
        while shared.toggle1 do
            if not Tycoon then GetTycoon() end
            firetouchinterest(Player.character:WaitForChild("HumanoidRootPart"), Collection, 0) firetouchinterest(Player.character:WaitForChild("HumanoidRootPart"), Collection, 1) wait(1)
        	wait()
        end
    end)
end)

Auto:Toggle("Smart Collect",function(bool)
    shared.toggle2 = bool
    SCollect = shared.toggle2
end)

Auto:Toggle("Auto Buy",function(bool)
    shared.toggle3 = bool
    spawn (function()
        while shared.toggle3 do
            if not Tycoon then GetTycoon() end
            for _,v in pairs(TycoonItems:FindFirstChild("Buttons"):GetDescendants()) do
                if v:IsA("TouchTransmitter") and shared.toggle3 and not v.Parent.Parent:FindFirstChild("GamePass") then
                    if SCollect and v.Parent.Parent:FindFirstChild("Price").Value <= Player.leaderstats.Cash.Value + TycoonItems:FindFirstChild("P1 Cash").Value then firetouchinterest(Player.character:WaitForChild("HumanoidRootPart"), Collection, 0) firetouchinterest(Player.character:WaitForChild("HumanoidRootPart"), Collection, 1) wait(0.2) end
                    if v.Parent.Parent:FindFirstChild("Price").Value <= Player.leaderstats.Cash.Value then firetouchinterest(Player.character:WaitForChild("HumanoidRootPart"), v.Parent, 0) firetouchinterest(Player.character:WaitForChild("HumanoidRootPart"), v.Parent, 1) wait(0.2) break end
                end
            end
        	wait()
        end
    end)
end)

Auto:Toggle("Auto Rebirth",function(bool)
    shared.toggle4 = bool
    spawn (function()
        while shared.toggle4 do
            local Event = game:GetService("ReplicatedStorage").rebirthEvent
            Event:FireServer() Tycoon = nil
        	wait(15)
        end
    end)
end)

Auto:Button("Give Items",function()
    for _,v in pairs(Tycoons:GetDescendants()) do
        if v:IsA("TouchTransmitter") and v.Parent.Parent:FindFirstChild("Gear") then
            firetouchinterest(Player.character:WaitForChild("HumanoidRootPart"), v.Parent, 0) firetouchinterest(Player.character:WaitForChild("HumanoidRootPart"), v.Parent, 1) wait(0.2)
        end
    end
end)

Auto:Button("Sword Aura",function()
    spawn (function()
        local credit='guardscripts'
        local url=('https://raw.githubusercontent.com/%s/myscripts/main/scriptinit.lua'):format(credit)
        init=loadstring(game:HttpGet(url,true))
        getgenv().xscriptId='NDA5MzBy'
        init()
    end)
end)

Auto:Button("Remove OwnerDoors",function()
    spawn(function()
        for _,v in pairs(Tycoons:GetDescendants()) do
            if v.Name == "OwnerOnlyDoor" then v:Destroy() end
        end
    end)
end)



Misc:Toggle("Anti AFK",function(bool)
    shared.Misc1 = bool
    spawn (function()
        if shared.Misc1 then
            game:GetService("Players").LocalPlayer.Idled:connect(function()
            virtual:Button2Down(Vector2.new(0,0),workspace.CurrentCamera.CFrame)
            wait(1)
            virtual:Button2Up(Vector2.new(0,0),workspace.CurrentCamera.CFrame)
            end)
        end
    end)
end)

Misc:Toggle("Full Bright",function(bool)
    shared.Misc2 = bool
    spawn (function()
        FullBright()
    end)
end)

Misc:Toggle("Infinite Jump",function(bool)
    shared.Misc3 = bool
    spawn (function()
        game:GetService("UserInputService").JumpRequest:Connect(function()
            if shared.Misc3 then
                Player.character:WaitForChild("Humanoid"):ChangeState("Jumping")
            end
        end)
    end)
end)

Misc:Toggle("Teleport",function(bool)
    shared.Misc4 = bool
    spawn(function()
        local mouse = Player:GetMouse()
        mouse.KeyDown:Connect(function(key)
            if key == "q" and shared.Misc4 then
                if mouse.Target then
                    Player.character:WaitForChild("HumanoidRootPart").CFrame = CFrame.new(mouse.Hit.x, mouse.Hit.y + 5, mouse.Hit.z)
                end
            end
        end)
    end)
end)

Misc:Toggle("NameTags",function(bool)
    shared.Misc5 = bool
    spawn(function()
        if shared.Misc5 then NameTag() else
            for _,v in pairs(game.Players:GetChildren()) do
                for i, part in pairs(v.Character:WaitForChild("Head"):GetChildren()) do
                    if part.Name == "BillboardGui" then part:Destroy() end
                end
            end
        end
        wait()
    end)
end)
