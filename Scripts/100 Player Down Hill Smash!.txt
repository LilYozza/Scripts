local function checkPart(_part)
    if _part and _part.Name == "Boulder" and _part:IsA("BasePart") then

        local p1 = _part:WaitForChild("TouchInterest", 2)
        local p2 = _part:WaitForChild("Hitbox", 2)
        --local p3 = _part:WaitForChild("TouchInterest", 2)

        _part.CanCollide = false
        if p1 then p1:Destroy() end
        if p2 then p2:Destroy() end
        --if p3 then p3:Destroy() end


    end
end

local bCon
Auto:Toggle("Easy Boulder",function(bool)
    shared.ToggleEB=bool
    spawn(function()
        if shared.ToggleEB then
            bCon = Workspace.ChildAdded:Connect(function(_part)
                checkPart(_part)
            end)

            for _,v in pairs(Workspace:GetChildren()) do
                checkPart(v)
            end
        else
            if bCon then bCon:Disconnect() end
        end
    end)
end)
