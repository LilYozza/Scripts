--Game Link: https://www.roblox.com/games/28382335/Survival-Zombie-Tycoon

local library = loadstring(game:HttpGet(('https://raw.githubusercontent.com/LilYozza/Scripts/main/Misc/UILibrary.txt')))()
local FullBright = loadstring(game:HttpGet('https://raw.githubusercontent.com/LilYozza/Scripts/main/Misc/FullBright.txt'))
local ESP = loadstring(game:HttpGet('https://raw.githubusercontent.com/LilYozza/Scripts/main/Misc/ESP.txt'))

local Player = game.Players.LocalPlayer
local character = Player.Character or Player.CharacterAdded:Wait()
local root = character:WaitForChild("HumanoidRootPart")
local virtual = game:GetService("VirtualUser")

local Tycoons = workspace:FindFirstChild("Tycoons")
local Tycoon = nil

local w = library:CreateWindow("Main")
local Auto = w:CreateFolder("Auto")
local Misc = w:CreateFolder("Misc")

for key, value in pairs(Tycoons:GetDescendants()) do
    if value.Name == "Owner" and value.Value == Player then
        Tycoon = value.Parent.Parent
        break
    end
end
Auto:Toggle("Auto Clicker",function(bool)
    shared.toggle1 = bool
    spawn (function()
        while shared.toggle1 do
            pcall(function()
                for key, value in pairs(Tycoon:GetDescendants()) do
                    if value.Name == "ClickDetector" and value:FindFirstAncestor("Droppers") then
                        fireclickdetector(value)
                        wait(0.4)
                    end
                    if value.Name == "ClickDetector" and value:FindFirstAncestor("Droppers") then
                        fireclickdetector(value)
                        wait(0.4)
                    end
                end
                wait()
            end)
        end
    end)
end)

Auto:Toggle("Auto Collect",function(bool)
    shared.toggle2 = bool
    spawn (function()
        while shared.toggle2 do
            pcall(function()
                for key, value in pairs(Tycoon:GetDescendants()) do
                    if value.Name == "TouchInterest" and value:FindFirstAncestor("EnergyCollector") then
                        firetouchinterest(Player.character:WaitForChild("HumanoidRootPart"), value.Parent, 0)
                        firetouchinterest(Player.character:WaitForChild("HumanoidRootPart"), value.Parent, 1)
                    end
                    if value.Name == "TouchInterest" and value:FindFirstAncestor("CashCollector") then
                        firetouchinterest(Player.character:WaitForChild("HumanoidRootPart"), value.Parent, 0)
                        firetouchinterest(Player.character:WaitForChild("HumanoidRootPart"), value.Parent, 1)
                    end
                end
                wait(1)
            end)
        end
    end)
end)

Auto:Toggle("Auto Buy",function(bool)
    shared.toggle3 = bool
    spawn (function()
        while shared.toggle3 do
            pcall(function()
                local Buttons = Tycoon:FindFirstChild("Buttons")
                for key, value in pairs(Buttons:GetDescendants()) do
                    if value.Name == "TouchInterest" and value.Parent.Parent.Parent.Name ~= "Robux" then
                        Cost = value.Parent.Parent:FindFirstChild("PriceValues")
                        if Cost:FindFirstChild("Cash") then Cash = Cost:FindFirstChild("Cash").value else Cash = 0 end
                        if Cost:FindFirstChild("Energy") then Energy = Cost:FindFirstChild("Energy").value else Energy = 0 end
                        if Cost:FindFirstChild("Rebirths") then Rebirths = Cost:FindFirstChild("Rebirths").value else Rebirths = 0 end
                        
                        if Cash <= Player.leaderstats.Cash.Value and Energy <= Player.leaderstats.Energy.Value and Rebirths <= Player.leaderstats.Rebirths.Value then
                            firetouchinterest(Player.character:WaitForChild("HumanoidRootPart"), value.Parent, 0)
                            firetouchinterest(Player.character:WaitForChild("HumanoidRootPart"), value.Parent, 1)
                            wait(1)
                        end
                    end
                end
                wait()
            end)
        end
    end)
end)

Auto:Toggle("Auto Rebirth",function(bool)
    shared.toggle4 = bool
    spawn (function()
        while shared.toggle4 do
            pcall(function()
                Configuration = Tycoon:FindFirstChild("Configuration")
                RebirthCooldownStatus = Configuration:FindFirstChild("RebirthCooldownStatus").Value
                ButtonProgress = Configuration:FindFirstChild("ButtonProgress").Value
                if not RebirthCooldownStatus and ButtonProgress == 1 then
                    local Event = game:GetService("ReplicatedStorage").RemoteEvents.Rebirth
                    Event:FireServer()
                end
                wait(1)
            end)
        end
    end)
end)

Auto:Toggle("Auto Heal",function(bool)
    shared.toggle5 = bool
    spawn (function()
        while shared.toggle5 do
            pcall(function()
                for key, value in pairs(Tycoon:GetDescendants()) do
                    if value.Name == "TouchPart" and value:FindFirstAncestor("HealPad") and Player.character:WaitForChild("Humanoid").Health < 10 then
                        firetouchinterest(Player.character:WaitForChild("HumanoidRootPart"), value, 0)
                        firetouchinterest(Player.character:WaitForChild("HumanoidRootPart"), value, 1)
                        wait()
                    end
                end
                wait()
            end)
        end
    end)
end)



Misc:Toggle("Anti AFK",function(bool)
    shared.Misc1 = bool
    spawn (function()
        if shared.Misc1 then
            game:GetService("Players").LocalPlayer.Idled:connect(function()
            virtual:Button2Down(Vector2.new(0,0),workspace.CurrentCamera.CFrame)
            wait(1)
            virtual:Button2Up(Vector2.new(0,0),workspace.CurrentCamera.CFrame)
            end)
        end
    end)
end)

Misc:Toggle("Full Bright",function(bool)
    shared.Misc2 = bool
    spawn (function()
        FullBright()
    end)
end)

Misc:Toggle("Infinite Jump",function(bool)
    shared.Misc3 = bool
    spawn (function()
        game:GetService("UserInputService").JumpRequest:Connect(function()
            if shared.Misc3 then
                Player.character:WaitForChild("Humanoid"):ChangeState("Jumping")
            end
        end)
    end)
end)

Misc:Toggle("Teleport",function(bool)
    shared.Misc4 = bool
    spawn(function()
        local mouse = Player:GetMouse()
        mouse.KeyDown:Connect(function(key)
            if key == "q" and shared.Misc4 then
                if mouse.Target then
                    Player.character:WaitForChild("HumanoidRootPart").CFrame = CFrame.new(mouse.Hit.x, mouse.Hit.y + 5, mouse.Hit.z)
                end
            end
        end)
    end)
end)

local Speed = Misc:Slider("Slider",{
    ['default'] = 16;
    ['min'] = 0;
    ['max'] = 100;
    ['precise'] = true;
},function(value)
    game:GetService("Players").LocalPlayer.Character.Humanoid.WalkSpeed = value
    local Players = game:GetService("Players")
end)

Misc:Button("3rd Person",function()
    game:GetService('Players').LocalPlayer.CameraMode = 'Classic'
    Player.CameraMaxZoomDistance = 150
    Player.CameraMinZoomDistance = 0
end)
Misc:Button("ESP",function()
    ESP()
end)


Misc:DestroyGui()
