local lastSnapshots = {}
local function poll(_stock)
    local foundUpdate = false
    for name, stock in pairs(_stock) do
        local ref = stock.Reference 
        if ref and ref[stock.Time] then
            local CurrentTime = ref[stock.Time]
            if not lastSnapshots[name] or lastSnapshots[name] ~= CurrentTime then
                foundUpdate = true
            end
            lastSnapshots[name] = CurrentTime
        end
    end
    return foundUpdate
end
local function updateStock(_stock)
    task.wait(2) -- Gonna Wait for Slow Stock Updates
    for name, stock in pairs(_stock) do
        local ref = stock.Reference

        if ref then
            -- Default stock reference
            stock.Stock = ref.Stocks

            -- Adds Seed
            if ref[stock.Time] then
                stock.Seed = ref[stock.Time]
            end

            if name == "TravelingMerchantShopStock" then
                stock.Type = ALLDATA.TravelingMerchantShopStock.MerchantType
                stock.LastUpdated = ALLDATA.TravelingMerchantShopStock.MerchantStartTime
                stock.Seed = ALLDATA.TravelingMerchantShopStock.MerchantStartTime

            elseif name == "CosmeticStock" then
                stock.LastUpdated = ALLDATA.CosmeticStock.CosmeticSeed
                stock.Seed = ALLDATA.CosmeticStock.CosmeticSeed

                -- Reuse table instead of recreating every update
                local t = stock.Stock

                if type(t) ~= "table" or t == ref.Stocks then
                    t = {}
                else
                    table.clear(t)
                end

                for _, v in pairs(ref) do
                    if type(v) == "table" then
                        for i, k in pairs(v) do
                            if type(k) == "table" and k.Stock then
                                local newStock = t[i]
                                if type(newStock) ~= "table" then
                                    newStock = {}
                                    t[i] = newStock
                                else
                                    table.clear(newStock)
                                end
                                for key, value in pairs(k) do
                                    newStock[key] = value
                                end
                                newStock.MaxStock = k.Stock
                            end
                        end
                    end
                end

                stock.Stock = t

            else
                stock.LastUpdated = os.time()
            end
        end
    end
end

local function getRandomPos()

    -- randomly choose one of the two parts
    local chosenPart = (math.random(1, 2) == 1) and point1 or point2

    local size = chosenPart.Size
    local cf = chosenPart.CFrame

    -- random offset inside the part
    local x = (math.random() - 0.5) * size.X
    --local y = (math.random() - 0.5) * size.Y
    local y = 0
    local z = (math.random() - 0.5) * size.Z

    -- convert local offset to world position
    return (cf * CFrame.new(x, y, z)).Position
end


local function checkPlantInventory(_plant, _mutations, _strict)
    for _, v in pairs(ALLDATA.InventoryData) do
        local data = v.ItemData
        if v.ItemType == "Holdable" and data and (not _plant or data.ItemName:lower() == _plant:lower()) and not data.IsFavorite then
            local str, matches = (data.MutationString or ""):lower(), 0
            for _, m in ipairs(_mutations or {}) do
                if str:find(m:lower(), 1, true) then matches = matches + 1 end
            end
            local valid = _strict and (matches == #_mutations) or (not _strict and matches > 0)
            if not _mutations or valid then return v, _ end
        end
    end
end
local function checkPlant(_plant, _name, _mutations, _strict, _size)
    -- 1. Base Filters
    _size = tonumber(_size) or nil
    if not _plant:HasTag("Harvestable") or _plant:GetAttribute("Favorited") then 
        return 
    end
    if _name and _plant.Name:lower() ~= _name:lower() then 
        return 
    end

    local weight = _plant:FindFirstChild("Weight")
    if _size and (not weight or _size > weight.Value) then
        --print("Failed Weight", weight, _size, weight.Value )
        return
    end

    if _mutations and #_mutations > 0 then
        -- Detect if "any" is in the mutation list
        local isAny = false
        for _, m in ipairs(_mutations) do
            if m:lower() == "any" then
                isAny = true
                break
            end
        end

        local matches = 0
        if isAny then
            -- Count any true attribute
            for _, value in pairs(_plant:GetAttributes()) do
                if value == true then
                    matches = matches + 1
                    if not _strict then break end
                end
            end
        else
            -- Count only listed mutations
            for _, m in ipairs(_mutations) do
                if _plant:GetAttribute(m) == true then
                    matches = matches + 1
                    if not _strict then break end
                end
            end
        end

        -- Validation Gate
        local isValid = _strict and (isAny and matches == #_plant:GetAttributes() or matches == #_mutations)
                            or (matches > 0)
        if not isValid then 
            return 
        end
    end

    return _plant
end

local function checkBackpack(_item, _amount,_mutation, _weight, wtype, _match)
    if not _item then return end

    local function matchTool(t)
        if not t or not t:IsA("Tool") then return end

        local cAttr = t:GetAttribute("c")
        local hAttr = t:GetAttribute("h")
        local eAttr = t:GetAttribute("e")
        local seedName = t:GetAttribute("Seed")
        local UUIDAttr = t:GetAttribute("UUID")
        local name = t.Name


        local weight = getWeight(name)
        wtype = wtype or true

        if _amount and (not eAttr or eAttr < _amount) then return end

        if (weight and _weight and ((wtype and weight >= _weight) or (not wtype and weight <= _weight))) or not _weight then
            if (_mutation and t:GetAttribute(_mutation)) or not _mutation then
                if (hAttr and hAttr == _item) or (cAttr and cAttr == _item) or (UUIDAttr and UUIDAttr == _item:upper()) or (name and (name == _item or (not _match and name:lower():match(_item:lower())))) then
                    return t, cAttr
                end
            end
        end
    end

    local tool, c = matchTool(Player.Character:FindFirstChildOfClass("Tool"))
    if tool then return tool, c end

    for _, v in pairs(Player.Backpack:GetChildren()) do
        local tool, c = matchTool(v)
        if tool then return tool, c end
    end
end

local function getFruit(_amount, _blacklist, _mutationList)
    local index, t = 1, {}
    local harvestablePrompts = CollectionService:GetTagged("CollectPrompt")
    --print("Found " .. #harvestablePrompts .. " harvestable fruits.")
    for _, prompt in ipairs(harvestablePrompts) do
        if _amount and index > _amount then break end
        local plant = (prompt.Parent and prompt.Parent.Parent) or nil
        if (plant and plant:IsDescendantOf(Plants_Physical) and prompt.Enabled == true and plant:HasTag("Harvestable")) then
            local DoneGrowTime = plant:GetAttribute("DoneGrowTime")
            if not DoneGrowTime or (os.time() > DoneGrowTime) then
                if not _blacklist or not table.find(_blacklist, plant.Name) then
                    local cPlant = checkPlant(plant, nil, _mutationList, false)
                    if cPlant then
                        t[index] = plant
                        index = index + 1
                    end
                end
            end
        end
    end
    return t
end

local function buyAllStock(_type)
    local type = _type.name
    local _frame = _type.path:WaitForChild("Frame")
    local list = _frame:WaitForChild("ScrollingFrame")
    for _,v in pairs(list:GetChildren())do
        local mFrame = v:FindFirstChild("Main_Frame")
        if v:IsA("Frame") and mFrame then
            local info = getStockInfo(mFrame)
            if info and info.stock and info.price and info.stock > 0 and Sheckles() >= info.price then
                if type == "seeds" then
                    buySeed(info.name, info.stock)
                elseif type == "gears" then
                    buyGear(info.name, info.stock)
                elseif type == "eggs" then
                    buyEgg(info.name, info.stock)
                elseif type == "merchant" then
                    buyMerchant(info.name, info.stock)
                end
            end
        end
    end
end

local function buyDaily()

    local seeds = {}
    for _,v in pairs(mFrame:GetChildren())do

            if info.stock > 0 and Sheckles() >= info.price then
                buyDailySeed(info.name, info.stock)
            end
        end
        if v.Name == "Seeds" then
            for __,_v in pairs(v:GetChildren())do

                    if info.stock > 0 and Sheckles() >= info.price then
                        buyDailySeed(info.name, info.stock)
                    end
                end
            end
        end
    end

local function ultraLow(i)
	local o = i
	if o:IsA("ParticleEmitter") or o:IsA("Trail") or o:IsA("Beam") or o:IsA("PointLight") or o:IsA("SpotLight") or o:IsA("SurfaceLight") then o.Enabled=false
	elseif o:IsA("BasePart") then o.CastShadow=false o.Reflectance=0 o.Material=Enum.Material.Plastic o.Transparency=1 o.CanCollide=false
	elseif o:IsA("MeshPart") then o.RenderFidelity=Enum.RenderFidelity.Performance
	elseif o:IsA("Decal") or o:IsA("Texture") then o.Transparency=1
	elseif o:IsA("Sound") then o.Volume=0 end

	for _,o in ipairs(i:GetDescendants()) do
		if o:IsA("ParticleEmitter") or o:IsA("Trail") or o:IsA("Beam") or o:IsA("PointLight") or o:IsA("SpotLight") or o:IsA("SurfaceLight") then o.Enabled=false
		elseif o:IsA("BasePart") then o.CastShadow=false o.Reflectance=0 o.Material=Enum.Material.Plastic o.Transparency=1 o.CanCollide=false
		elseif o:IsA("MeshPart") then o.RenderFidelity=Enum.RenderFidelity.Performance
		elseif o:IsA("Decal") or o:IsA("Texture") then o.Transparency=1
		elseif o:IsA("Sound") then o.Volume=0 end
	end
end

local function getCurrentPets()
    local t = {}
    for _,v in pairs(PetDataTable) do
        if v.PetType and not table.find(t, v.PetType) then
            table.insert(t, v.PetType)
        end
    end
    t = SortTableByName(t)
    return t
end

local function getPetCount(_pet)
    local count = 0
    for _,v in pairs(PetDataTable)do
        if v.PetType and v.PetType:lower() == _pet:lower() then
            count = count + 1
        end
    end
    return count
end

local function addToTable(_tbl, _item)
    if not table.find(_tbl, _item) then
        table.insert(_tbl, _item)
        print("Added:", sPetDelete)
    end
end

local function removetoTable(_tbl, _item)
    if table.find(_tbl, _item) then
        table.insert(_tbl, _item)
        print("Removed:", sPetDelete)
    end
end

PetMenu:Button("Add",function()
    spawn(function()
        if sPetDelete then
            if sPetDelete == "All" then
                for _,v in pairs(getCurrentPets())do
                    addToTable(sPetDeleteTable, v)
                end
            else
                addToTable(sPetDeleteTable, sPetDelete)
            end
        end
    end)
end)

PetMenu:Button("Remove",function()
    spawn(function()
        if sPetDelete then
            if sPetDelete == "All" then
                for _,v in pairs(getCurrentPets())do
                    removetoTable(sPetDeleteTable, v)
                end
            else
                removetoTable(sPetDeleteTable, sPetDelete)
            end
        end
    end)
end)
