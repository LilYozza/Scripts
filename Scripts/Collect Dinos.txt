local function getSlotInfo()
    local t = {}

    for _, slot in ipairs(EggSlots:GetChildren()) do
        if slot.Name:find("Slot") then
            local egg = slot:FindFirstChildWhichIsA("Tool") or nil
            local timerText = slot.TimerPart.SurfaceGui.TimerFrame.TimerText
            local text = timerText.Text

            t[slot] = {
                egg = egg,
                isReady = text == "Ready",
                timeLeft = (tonumber(text:match("(%d+)m")) or 0) * 60
                        + (tonumber(text:match("(%d+)s")) or 0),
                slot = slot
            }
        end
    end

    return t
end

local function isSlotsFull()
    local slots = getSlotInfo()
    for _,v in pairs(slots)do
        if not v.egg then
            return false
        end
    end
    return true
end

Auto:Button("Place Egg",function()
    spawn(function()
        startHatchingRemote:FireServer()
    end)
end)


Auto:Toggle("Auto Collect",function(bool)
    shared.ToggleAC=bool
    spawn(function()
        while shared.ToggleAC do
            for _,v in pairs(UnitSigns:GetChildren())do
                local btn = v:FindFirstChild("Button")
                if btn then
                    Touch(btn)
                end
            end
            task.wait(1)
        end
    end)
end)

local conIO
Auto:Toggle("Auto Place",function(bool)
    shared.ToggleAP=bool
    spawn(function()
        if shared.ToggleAP then
            conIO = Player:GetAttributeChangedSignal("holdingEgg"):Connect(function()
                local holdingEgg = Player:GetAttribute("holdingEgg")
                if holdingEgg and holdingEgg == true then
                    if not isSlotsFull() then
                        startHatchingRemote:FireServer()
                    end
                end
            end)
        else
            if conIO then conIO:Disconnect() conIO = nil end
        end
    end)
end)

Auto:Toggle("Auto Hatch",function(bool)
    shared.ToggleAH=bool
    spawn(function()
        while shared.ToggleAH do
            local slots = getSlotInfo()
            if slots then
                local lastPos
                for _,v in pairs(slots)do
                    if v.egg and v.isReady then
                        local slot = v.slot
                        local prox = slot:FindFirstChild("HatchEggPrompt")
                        if slot and prox then
                            if not lastPos then lastPos = GetPos() end
                            MoveTo(v.slot) task.wait(0.3)
                            fireproximityprompt(prox) task.wait(0.1)
                        end
                    end
                end
                if lastPos then MoveTo(lastPos) end
            end
            task.wait(1)
        end
    end)
end)
