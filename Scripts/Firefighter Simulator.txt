--Set Data

--Non Return Functions
local function GetPart(t) for _,v in pairs(t:GetChildren()) do if v:FindFirstChild("FireParts") then for num,p in pairs(v:FindFirstChild("FireParts"):GetChildren()) do if p:IsA("BasePart") then p.Size=Vector3.new(1,1,1) p.CFrame=Player.Character:WaitForChild("HumanoidRootPart").CFrame+Vector3.new(5,0,0) end end end end end
local function Touch(p,t) firetouchinterest(Player.Character:WaitForChild("HumanoidRootPart"),p,0)firetouchinterest(Player.Character:WaitForChild("HumanoidRootPart"),p,1)task.wait(t)end
local function MoveTo(p) local P=Player.Character:WaitForChild("HumanoidRootPart") P.Anchored=true local pos1=P.CFrame P.CFrame=p task.wait(5) P.CFrame=pos1 P.Anchored=false end

local ZoneLabel=Player.PlayerGui.Main.ZoneCompletionFrame
--Return Functions
local function CurrentStage() if ZoneLabel.TopTextLabel.Text:match("You made it") and not ZoneLabel.TopTextLabel.Text:match("World 1!") then if ZoneLabel.Percent.Text=="100.0 %" then return "nil" else return "Island Fantasy" end else if ZoneLabel.TopTextLabel.Text:match("World 1!") then return "Jungle" else local s1=string.split(ZoneLabel.TopTextLabel.Text, "in the ") local s2=string.split(s1[2]," to unlock") return s2[1] end end end

Auto:Toggle("Auto Complete",function(bool)
    shared.toggle1 = bool
    spawn (function() 
        while shared.toggle1 do
            Tool = game.Players.LocalPlayer.Backpack:FindFirstChildWhichIsA('Tool') or game.Players.LocalPlayer.Character:FindFirstChildWhichIsA('Tool')
            if Tool then Player.Character:WaitForChild("Humanoid"):EquipTool(Tool) local Holding = Tool.Configuration.Holding Holding.Value=1 local Damage = Tool.Configuration.Damage Damage.Value = 999999999 local Distance = Tool.Configuration.Distance Distance.Value =  999999999 local FuelUse = Tool.Configuration.FuelUse FuelUse.Value = -9999 local HitRadius = Tool.Configuration.HitRadius HitRadius.Value = 999999999 end
             local Bp=Player.Character:WaitForChild("TankBackpack")
            
            while Bp.Configuration.Fuel.Value <= 0 do Bp.Configuration.Fuel.Value=100 task.wait()end
            while not table.find(World2Zone, CurrentStage()) do GetPart(Zones) task.wait(3) end
            
            for _,v in pairs(World2Zone) do
                if CurrentStage() ~= "nil" and CurrentStage()==v then Touch(Teleporters:FindFirstChild(v).TouchPart, 2)
                repeat GetPart(Zones2) task.wait(1) until CurrentStage()==World2Zone[_+1] or CurrentStage()=="nil" end
            end
            if isRebirth then game:GetService("ReplicatedStorage").Remotes.RequestRebirth:InvokeServer() end
        	task.wait(5)
        end
    end)
end)

Auto:Toggle("Auto Rebirth",function(bool)
    spawn (function() isRebirth = bool end)
end)

Misc:Toggle("Infinite Jump",function(bool)
    shared.Misc3 = bool

end)

Misc:Toggle("Teleport",function(bool)
    shared.Misc4 = bool
    spawn(function()
        local mouse = Player:GetMouse()
        mouse.KeyDown:Connect(function(key)
            if key == "q" and shared.Misc4 then
                if mouse.Target then
                    Player.character:WaitForChild("HumanoidRootPart").CFrame = CFrame.new(mouse.Hit.x, mouse.Hit.y + 5, mouse.Hit.z)
                end
            end
        end)
    end)
end)

Misc:Toggle("NameTags",function(bool)
    shared.Misc5 = bool
    spawn(function()
        if shared.Misc5 then NameTag() else
            for _,v in pairs(game.Players:GetChildren()) do
                for i, part in pairs(v.Character:WaitForChild("Head"):GetChildren()) do
                    if part.Name == "BillboardGui" then part:Destroy() end
                end
            end
        end
        wait()
    end)
end)
