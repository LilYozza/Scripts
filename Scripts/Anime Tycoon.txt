local library = loadstring(game:HttpGet(('https://raw.githubusercontent.com/LilYozza/Scripts/main/Misc/UILibrary.txt')))()w = library:CreateWindow("Main")

local Players=game:GetService("Players")
local Player=game.Players.LocalPlayer

local Auto = w:CreateFolder("Auto")

local Tycoon, NilTycoon, SCollect, Collection
local Tycoons = workspace:FindFirstChild("Tycoons")

local function Touch(p,t)firetouchinterest(Player.Character:WaitForChild("HumanoidRootPart"),p,0)firetouchinterest(Player.Character:WaitForChild("HumanoidRootPart"),p,1)task.wait(t)end


local function GetTycoon()
    for _,v in pairs(Tycoons:GetDescendants()) do
        if v.Name == "Owner" and v.Value == Player.Name then Tycoon = v.Parent end
        if v.Name == "Owner" and v.Value == "None" and not NilTycoon then NilTycoon = v.Parent end
    end
    if not Tycoon then Tycoon = NilTycoon
        for _,v in pairs(Tycoon:GetDescendants()) do
            if v:IsA("TouchTransmitter") and v:FindFirstAncestor("Door") then
                firetouchinterest(Player.character:WaitForChild("HumanoidRootPart"), v.Parent, 0) firetouchinterest(Player.character:WaitForChild("HumanoidRootPart"), v.Parent, 1) wait()
                Player.character:WaitForChild("HumanoidRootPart").CFrame = v:FindFirstAncestor("Essential"):FindFirstChild("SpawnLocation").CFrame + Vector3.new(0,6,0)
            end
        end
    end
    for _,v in pairs(Tycoon:GetDescendants()) do if v:IsA("TouchTransmitter") and v:FindFirstAncestor("Collect") then Collection = v.Parent end end
end GetTycoon()

Auto:Toggle("Auto Collect",function(bool)
    shared.toggle1 = bool
    spawn (function()
        while shared.toggle1 do
            if not Tycoon then GetTycoon() end
            Touch(Collection,0.1)
        end
    end)
end)

Auto:Toggle("Smart Collect",function(bool)
    shared.toggle1 = bool
    SCollect = shared.toggle1
end)

Auto:Toggle("Auto Buy",function(bool)
    shared.toggle1 = bool
    spawn (function()
        while shared.toggle1 do
            if not Tycoon then GetTycoon() end
            for _,v in pairs(Tycoon:FindFirstChild("Buttons"):GetDescendants()) do
                if v:IsA("TouchTransmitter") and v.Parent.Transparency == 0 and not v.Parent.Parent:FindFirstChild("Gamepass") then
                    if SCollect and v.Parent.Parent:FindFirstChild("Price").Value <= Player.leaderstats.Cash.Value + Tycoon:FindFirstChild("CashToCollect").Value then
                       Touch(Collection,0.5)
                    end
                    if v.Parent.Parent:FindFirstChild("Price").Value <= Player.leaderstats.Cash.Value then
                        Touvh(v.Parent,0.5)
                    end
                end
            end
        	wait(0.5)
        end
    end)
end)
Auto:Button("Give Items",function()
    spawn(function()
        for _,v in pairs(Tycoons:GetDescendants()) do
            if v:IsA("TouchTransmitter") and v.Parent.Parent:FindFirstChild("Tool") then
            firetouchinterest(Player.character:WaitForChild("HumanoidRootPart"), v.Parent, 0) firetouchinterest(Player.character:WaitForChild("HumanoidRootPart"), v.Parent, 1) wait()
            end
        end
    end)
end)

Auto:Button("Remove Doors",function()
    spawn (function()
        for _,v in pairs(Tycoons:GetDescendants()) do
            if v.Name == "OwnerOnlyDoor" then v:Destroy() end
        end
    end)
end)
