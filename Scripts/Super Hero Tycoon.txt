local library = loadstring(game:HttpGet(('https://raw.githubusercontent.com/LilYozza/Scripts/main/Misc/UILibrary.txt')))()w = library:CreateWindow("Main")

local Players=game:GetService("Players")
local Player=game.Players.LocalPlayer

local Auto = w:CreateFolder("Auto")
local Tycoon, NilTycoon, SCollect, Collection
local Tycoons = workspace:FindFirstChild("Tycoons"):FindFirstChild("Tycoons")
local Crates = workspace:FindFirstChild("Crates")

local function Touch(p,t)firetouchinterest(Player.Character:WaitForChild("HumanoidRootPart"),p,0)firetouchinterest(Player.Character:WaitForChild("HumanoidRootPart"),p,1)task.wait(t)end
local function fromSuffixString(s) local suffixes = {"K", "M", "B", "T", "Q"} found = false if s == "0" then return 0 end if s:match("$") then s = string.gsub(s, '%$', "") end if s:match(",") then s = string.gsub(s, '%,', "") end if s:match("+") then s = string.gsub(s, '%+', "") end for i, value in ipairs(suffixes) do if s:match(value) then found = true end end if found == false then return tonumber(s) end local n, suffix = string.match(s, "(.*)(%a)$") if n and suffix then local i = table.find(suffixes, suffix) or 0 return tonumber(n) * math.pow(10, i * 3) end return tonumber(s) end
local function GetTycoon()
    for _,v in pairs(Tycoons:GetDescendants()) do
        if v.Name == "Owner" and v.Value == Player then Tycoon = v.Parent end
        if v.Name == "Owner" and v.Value == nil and not NilTycoon and not v:FindFirstAncestor("The Flash") then NilTycoon = v.Parent end
    end
    if not Tycoon then Tycoon = NilTycoon
        for _,v in pairs(Tycoon:GetDescendants()) do
            if v:IsA("TouchTransmitter") and v:FindFirstAncestor("Entrance") then
                Touch(v.Parent)
                Player.character:WaitForChild("HumanoidRootPart").CFrame = Tycoon:FindFirstChild("Essentials"):FindFirstChild("Spawn").CFrame + Vector3.new(0,6,0)
            end
        end
    end
    for _,v in pairs(Tycoon:GetDescendants()) do if v:IsA("TouchTransmitter") and v:FindFirstAncestor("Essentials") and v.Parent.Name == "Giver" then Collection = v.Parent end end
end GetTycoon()

Auto:Toggle("Auto Crate",function(bool)
    shared.toggle1 = bool
    spawn (function()
        while shared.toggle1 do
            for _,v in pairs(Crates:GetDescendants()) do
                if v:IsA("TouchTransmitter") and shared.toggle1 then
                    Touch(v.Parent)
                end
            end
        	task.wait(0.5)
        end
    end)
end)

Auto:Toggle("Auto Click",function(bool)
    shared.toggle2 = bool
    if not Tycoon then GetTycoon() end
    spawn (function()
        while shared.toggle2 do
            for _,v in pairs(Tycoon:GetDescendants()) do
                if v:IsA("ClickDetector") and v.Parent.Name == "clicker" and shared.toggle2 then fireclickdetector(v) wait(0.2) end
            end
        	wait()
        end
    end)
end)

Auto:Toggle("Auto Collect",function(bool)
    shared.toggle3 = bool
    if not Tycoon then GetTycoon() end
    spawn (function()
        if not Tycoon then GetTycoon() end
        while shared.toggle3 do
            Touch(Collection,1)
        end
    end)
end)

Auto:Toggle("Smart Collect",function(bool)
    shared.toggle4 = bool
    SCollect = shared.toggle4
end)

Auto:Toggle("Auto Buy",function(bool)
    shared.toggle5 = bool
    if not Tycoon then GetTycoon() end
    spawn (function()
        while shared.toggle5 do
            for _,v in pairs(Tycoon:FindFirstChild("Buttons"):GetDescendants()) do
                if v:IsA("TouchTransmitter") and v.Parent.Transparency == 0 and not v.Parent.Parent:FindFirstChild("Gamepass") and shared.toggle5 then
                    if SCollect and v.Parent.Parent:FindFirstChild("Price").Value <= fromSuffixString(Player.leaderstats.Cash.Value) + Tycoon:FindFirstChild("CurrencyToCollect").Value then Touch(Collection,0.5) end
                    if v.Parent.Parent:FindFirstChild("Price").Value <= fromSuffixString(Player.leaderstats.Cash.Value) then Touch(v.Parent,0.5) end
                end
            end
        	wait()
        end
    end)
end)

Auto:Button("Give Items",function()
    for _,v in pairs(Tycoons:GetDescendants()) do
        if v:IsA("TouchTransmitter") and v.Parent.Name == "Giver" then
            Touch(v.Parent,0.1)
        end
    end
end)

Auto:Button("Remove OwnerDoors",function()
    spawn (function()
        for _,v in pairs(Tycoons:GetDescendants()) do
            if v:IsA("TouchTransmitter") and v.Parent.Name == "Pipe" then v.Parent:destroy() end
        end
    end)
end)
