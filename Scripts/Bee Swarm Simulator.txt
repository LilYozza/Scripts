local Players = game:GetService("Players")

shared.pAmount = 1
Auto:Slider("Players",{
    min = 1; max = 6; precise = true;
},function(value)
    shared.pAmount = value
end)

Auto:Toggle("Player Check",function(bool)
    shared.ToggleHC=bool
    spawn(function()
        while shared.ToggleHC do
            if #Players:GetPlayers() > shared.pAmount and shared.ToggleHC then
                Player:Kick("Someone Joined.")
            end
            task.wait(1)
        end
    end)
end)
