local Player = game.Players.LocalPlayer
local character = Player.Character or Player.CharacterAdded:Wait()
local root = character:WaitForChild("HumanoidRootPart")

local Tycoons, Tycoon
local suffixes = {"K", "M", "B", "T", "Q"}

local w = library:CreateWindow("Main")
local Auto = w:CreateFolder("Auto")
local Misc = w:CreateFolder("Misc")

function CheckTycoon(part)
    if part:FindFirstAncestor(Tycoons.name) and part:FindFirstAncestor(Tycoon.name):FindFirstChild("Owner").Value == Player then 
        return 1 
    else 
        return 0 
    end
end

local function fromSuffixString(s)
    found = false
    if s:match("$") then s = string.gsub(s, '%$', "") end
    if s:match(",") then s = string.gsub(s, '%,', "") end
    for i, value in ipairs(suffixes) do
        if s:match(value) then found = true end
    end
    if found == false then return tonumber(s) end
	local n, suffix = string.match(s, "(.*)(%a)$")
	if n and suffix then
		local i = table.find(suffixes, suffix) or 0
		return tonumber(n) * math.pow(10, i * 3)
	end
	return tonumber(s)
end

function GetTycoon()
    for key, value in pairs(workspace:GetDescendants()) do
        if value.Name == "Owner" and value.Value == Player then
            Tycoon = value.Parent Tycoons = value.Parent.Parent.Parent
            break
        end
    end
    if Tycoon == nil then 
        for key, value in pairs(workspace:GetDescendants()) do
            if value.Name == "Owner" and value.Value == nil then
                claim = value.Parent:FindFirstChild("Entrance"):FindFirstChild("Claim"):FindFirstChild("Head")
                firetouchinterest(Player.character:WaitForChild("HumanoidRootPart"), claim, 0)
                firetouchinterest(Player.character:WaitForChild("HumanoidRootPart"), claim, 1)
                Tycoon = value.Parent Tycoons = value.Parent.Parent.Parent
                break
            end
        end   
    end
end
GetTycoon()

function SpawnAtTycoon()
    for i, part in pairs(Tycoons:FindFirstChild("Tycoons"):FindFirstChild(Tycoon.Name):GetDescendants()) do
        if part.Name == "Spawn" and CheckTycoon(part) then
            Player.character:WaitForChild("HumanoidRootPart").CFrame = part.CFrame
        end
    end
end

spawn(function() wait(3) SpawnAtTycoon())

Auto:Toggle("Auto Collect",function(bool)
    shared.toggle1 = bool
    spawn (function()
        while shared.toggle1 do
            pcall(function()
                    for key, value in pairs(Tycoons:FindFirstChild("Tycoons"):FindFirstChild(Tycoon.Name):GetDescendants()) do
                        if  value.Name == "GroupGiver" or value.Name == "Giver" and CheckTycoon(value) and value.Parent.Name == "Essentials" then
                            firetouchinterest(Player.character:WaitForChild("HumanoidRootPart"), value, 0)
                            firetouchinterest(Player.character:WaitForChild("HumanoidRootPart"), value, 1)
                        end
                    end
                wait(1)
            end)
        end
    end)
end)

Auto:Toggle("Auto Buy",function(bool)
    shared.toggle2 = bool
    spawn (function()
        while shared.toggle2 do
            pcall(function()
                for key, value in pairs(Tycoons:FindFirstChild("Tycoons"):FindFirstChild(Tycoon.Name):FindFirstChild("Buttons"):GetDescendants()) do
                    if value.Name == "Buton" and CheckTycoon(value) and value.Transparency == 0 then
                        price = value:FindFirstChild("BillboardGui"):FindFirstChild("price").text    
                        if not price:match("R") then cost = fromSuffixString(price) end
                        if type(cost) == "number" and Player.leaderstats.Cash.value >= cost and CheckTycoon(value) then
                            firetouchinterest(Player.character:WaitForChild("HumanoidRootPart"), value, 0)
                            firetouchinterest(Player.character:WaitForChild("HumanoidRootPart"), value, 1)
                            wait()
                        end
                    end
                end
                wait()
            end)
        end
    end)
end)

Auto:Toggle("Auto Rebirth",function(bool)
    shared.toggle1 = bool
    spawn (function()
        while shared.toggle1 do
            pcall(function()
                for key, value in pairs(Tycoons:FindFirstChild("Tycoons"):FindFirstChild(Tycoon.Name):GetDescendants()) do
                    if CheckTycoon(value) and value.Name == "PP" then
                        Player.character:WaitForChild("HumanoidRootPart").CFrame = value.Parent.CFrame
                        wait(1)
                        fireproximityprompt(value)
                        wait(5)
                        GetTycoon()
                        wait(1)
                        SpawnAtTycoon()
                    end
                end
                wait(60)
            end)
        end
    end)
end)
