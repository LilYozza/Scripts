local function fromSuffixString(s)
    found = false
    if s == "0" then return 0 end
    if s:match("$") then s = string.gsub(s, '%$', "") end
    if s:match(",") then s = string.gsub(s, '%,', "") end
    if s:match("+") then s = string.gsub(s, '%+', "") end
    s = string.gsub(s, '%(', "")
    s = string.gsub(s, '%)', "")
    for i, value in ipairs(suffixes) do
        if s:match(value) then found = true end
    end
    if found == false then return tonumber(s) end
	local n, suffix = string.match(s, "(.*)(%a)$")
	if n and suffix then
		local i = table.find(suffixes, suffix) or 0
		return tonumber(n) * math.pow(10, i * 3)
	end
	return tonumber(s)
end
function GetPrice(p) for _,v in pairs(p:FindFirstChild("Purchase"):GetChildren()) do if v.Name == "Price" then return fromSuffixString(v.text) end end end

Auto:Toggle("Auto Upgrade",function(bool)
    shared.toggle1 = bool
    spawn (function()
        while shared.toggle1 do
            for _,v in pairs(Upgrades:GetChildren()) do
                if v:IsA("Frame") and not v:FindFirstChild("Purchase"):FindFirstChild("Price").text:match("MAXED")  and shared.toggle1 then
                    if GetPrice(v) <= fromSuffixString(Player.leaderstats.Cash.Value) then
                        local Button = v:FindFirstChild("Purchase") firesignal(Button["MouseButton1Down"]) wait() firesignal(Button["MouseButton1Up"])
                    end
                end
            end
        	wait()
        end
    end)
end)
