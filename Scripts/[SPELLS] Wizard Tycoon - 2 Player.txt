local library = loadstring(game:HttpGet(('https://raw.githubusercontent.com/LilYozza/Scripts/main/Misc/UILibrary.txt')))()
local FullBright = loadstring(game:HttpGet('https://raw.githubusercontent.com/LilYozza/Scripts/main/Misc/FullBright.txt'))
local ESP = loadstring(game:HttpGet('https://raw.githubusercontent.com/LilYozza/Scripts/main/Misc/ESP.txt'))

local Player = game.Players.LocalPlayer
local character = Player.Character or Player.CharacterAdded:Wait()
local root = character:WaitForChild("HumanoidRootPart")
local virtual = game:GetService("VirtualUser")

local Tycoon, NilTycoon, SCollect
local Tycoons = workspace:FindFirstChild("berezaa's Tycoon Kit")

local w = library:CreateWindow("Main")
local Auto = w:CreateFolder("Auto")
local Misc = w:CreateFolder("Misc")

function GetTycoon()
    for _,v in pairs(Tycoons:GetDescendants()) do
        if v:isA("ObjectValue") and v.Value == Player then
            Tycoon = v.Parent break
        end
        if v:isA("ObjectValue") and v.Value == nil and not NilTycoon then
            NilTycoon = v.Parent
        end
    end
    if not Tycoon then Tycoon = NilTycoon
        firetouchinterest(Player.character:WaitForChild("HumanoidRootPart"), Tycoon:FindFirstChild("Gate"):FindFirstChild("Head"), 0)
        firetouchinterest(Player.character:WaitForChild("HumanoidRootPart"), Tycoon:FindFirstChild("Gate"):FindFirstChild("Head"), 1) wait(1)
        Player.character:WaitForChild("HumanoidRootPart").CFrame = Tycoon:FindFirstChild("Gate"):FindFirstChild("Head").CFrame
    end
end

GetTycoon()

Auto:Toggle("Auto Clicker",function(bool)
    shared.toggle1 = bool
    spawn (function()
        if not Tycoon then GetTycoon() end
        while shared.toggle1 do
            for _,v in pairs(Tycoon.Parent:GetDescendants()) do
                if v:isA("ClickDetector") and v:FindFirstAncestor("Mine") and shared.toggle1 then
                    fireclickdetector(v)
                    wait(0.5)
                end
            end
        	wait()
        end
    end)
end)

Auto:Toggle("Auto Collect",function(bool)
    shared.toggle2 = bool
    spawn (function()
        if not Tycoon then GetTycoon() end
        while shared.toggle2 do
            for _,v in pairs(Tycoon.Parent:FindFirstChild("Essentials"):GetDescendants()) do
                if v:isA("TouchTransmitter") and v.Parent.Name:match("Giver") and shared.toggle2 then
                    firetouchinterest(Player.character:WaitForChild("HumanoidRootPart"), v.Parent, 0)
                    firetouchinterest(Player.character:WaitForChild("HumanoidRootPart"), v.Parent, 1) wait(1) break
                end
            end
        	wait()
        end
    end)
end)

Auto:Toggle("Smart Collect",function(bool)
    shared.toggle3 = bool
    SCollect = shared.toggle3
end)

Auto:Toggle("Auto Buy",function(bool)
    shared.toggle4 = bool
    spawn (function()
        if not Tycoon then GetTycoon() end
        while shared.toggle4 do
            for _,v in pairs(Tycoon:FindFirstChild("Buttons"):GetDescendants()) do
                if v:IsA("TouchTransmitter") and v.Parent.Transparency == 0 and v.Parent.Parent:FindFirstChild("Price") and shared.toggle4 then
                    if SCollect and v.Parent.Parent:FindFirstChild("Price").Value <= Player.leaderstats.Cash.Value + tonumber(Tycoon:FindFirstChild("Cash to collect: $0"):FindFirstChild("Head"):FindFirstChild("SurfaceGui"):FindFirstChild("TextLabel").Text:split("$")[2])  then
                        firetouchinterest(Player.character:WaitForChild("HumanoidRootPart"), Tycoon.Parent:FindFirstChild("Essentials"):FindFirstChild("Giver"), 0)
                        firetouchinterest(Player.character:WaitForChild("HumanoidRootPart"), Tycoon.Parent:FindFirstChild("Essentials"):FindFirstChild("Giver"), 1) wait()
                        firetouchinterest(Player.character:WaitForChild("HumanoidRootPart"), Tycoon.Parent:FindFirstChild("Essentials"):FindFirstChild("Giver2"), 0)
                        firetouchinterest(Player.character:WaitForChild("HumanoidRootPart"), Tycoon.Parent:FindFirstChild("Essentials"):FindFirstChild("Giver2"), 1) wait(1)
                    end
                    if v.Parent.Parent:FindFirstChild("Price").Value <= Player.leaderstats.Cash.Value then
                        firetouchinterest(Player.character:WaitForChild("HumanoidRootPart"), v.Parent, 0)
                        firetouchinterest(Player.character:WaitForChild("HumanoidRootPart"), v.Parent, 1) wait(0.5)
                    end
                    
                end
            end
        	wait()
        end
    end)
end)

Misc:Toggle("Teleport",function(bool)
    shared.Misc4 = bool
    spawn(function()
        local mouse = Player:GetMouse()
        mouse.KeyDown:Connect(function(key)
            if key == "q" and shared.Misc4 then
                if mouse.Target then
                    Player.character:WaitForChild("HumanoidRootPart").CFrame = CFrame.new(mouse.Hit.x, mouse.Hit.y + 5, mouse.Hit.z)
                end
            end
        end)
    end)
end)
