local library = loadstring(game:HttpGet(('https://raw.githubusercontent.com/LilYozza/Scripts/main/Misc/UILibrary.txt')))()w = library:CreateWindow("Main")

local Players=game:GetService("Players")
local Player=game.Players.LocalPlayer

local Auto = w:CreateFolder("Auto")

local Tycoon, NilTycoon, SCollect, PTeam
local Tycoons = workspace:FindFirstChild("berezaa's Tycoon Kit")
repeat task.wait()until Player.Team

--Non Return Functions
local function Touch(p,t)firetouchinterest(Player.Character:WaitForChild("HumanoidRootPart"),p,0)firetouchinterest(Player.Character:WaitForChild("HumanoidRootPart"),p,1)task.wait(t)end
local function MoveTo(p,offset)Player.Character:WaitForChild("HumanoidRootPart").CFrame=p.CFrame+offset end

--Checking/Getting Data
local function GetTeam()for _,v in pairs(Tycoons:GetChildren())do if v.Name:lower():match(Player.Team.Name:lower()) then return v end end end
local function GetTycoon()for _,v in pairs(PTeam:GetChildren())do if v:FindFirstChild("Owner") and v.Owner.Value==Player then Tycoon=v break end end end
local function ClaimTycoon()for _,v in pairs(Tycoons:GetDescendants())do if v:FindFirstChild("Owner") and v.Owner.Value==nil then return v end end end
PTeam=GetTeam()if PTeam then GetTycoon()else NilTycoon=ClaimTycoon()Touch(NilTycoon.Gate.Head)MoveTo(NilTycoon.Gate.Head,Vector3.new(0,4,0))task.wait(1)PTeam=GetTeam()GetTycoon()end


Auto:Toggle("Auto Clicker",function(bool)
    shared.toggle1 = bool
    spawn (function()
        while shared.toggle1 do
            for _,v in pairs(PTeam:GetChildren())do
                if v:FindFirstChild("PurchasedObjects") and v.PurchasedObjects:FindFirstChild("Mine") then
                    fireclickdetector(v.PurchasedObjects.Mine.Button.ClickDetector)
                end
            end
        	task.wait()
        end
    end)
end)

Auto:Toggle("Auto Collect",function(bool)
    shared.toggle2 = bool
    spawn (function() while shared.toggle2 do Touch(PTeam.Essentials.Giver)Touch(PTeam.Essentials.Giver2,1)task.wait()end end)
end)

Auto:Toggle("Smart Collect",function(bool)
    shared.toggle3 = bool SCollect = shared.toggle3
end)

Auto:Toggle("Auto Buy",function(bool)
    shared.toggle4 = bool
    spawn (function()
        if not Tycoon then GetTycoon() end
        while shared.toggle4 do
            for _,v in pairs(Tycoon:FindFirstChild("Buttons"):GetDescendants()) do
                if v:IsA("TouchTransmitter") and v.Parent.Transparency == 0 and v.Parent.Parent:FindFirstChild("Price") and shared.toggle4 then
                    if SCollect and v.Parent.Parent:FindFirstChild("Price").Value <= Player.leaderstats.Cash.Value + tonumber(Tycoon:FindFirstChild("Cash to collect: $0"):FindFirstChild("Head"):FindFirstChild("SurfaceGui"):FindFirstChild("TextLabel").Text:split("$")[2])  then
                        Touch(PTeam.Essentials.Giver)Touch(PTeam.Essentials.Giver2,1)
                    end
                    if v.Parent.Parent:FindFirstChild("Price").Value <= Player.leaderstats.Cash.Value then
                        Touch(v.Parent,0.5)
                    end
                    
                end
            end
        	task.wait()
        end
    end)
end)

Auto:Button("Fire Cannons",function()
    spawn (function()
        for _,v in pairs(PTeam:GetChildren())do
            if v:FindFirstChild("PurchasedObjects")then
                for i,p in pairs(v:GetDescendants())do
                    if p:IsA("ClickDetector")then fireclickdetector(p)end
                end
            end
        end
    end)
end)

Auto:Button("Give Items",function()
    spawn(function()
        for _,v in pairs(Tycoons:GetDescendants()) do
            if v:IsA("TouchTransmitter") and v.Parent.Name:match("Staff") then
                Touch(v.Parent)
            end
            if v:IsA("TouchTransmitter") and v.Parent.Name:match("Broom") then
                Touch(v.Parent)
            end
        end
    end)
end)
Auto:Button("Give Potion",function()
    spawn(function()
        for _,v in pairs(Tycoons:GetDescendants()) do
            if v:IsA("ClickDetector") and v.Parent.Name:match("Potion") then
                fireclickdetector(v) break
            end
        end
    end)
end)
