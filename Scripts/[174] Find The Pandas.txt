local library = loadstring(game:HttpGet(('https://raw.githubusercontent.com/LilYozza/Scripts/main/Misc/UILibrary.txt')))()w = library:CreateWindow("Main")

local Players=game:GetService("Players")
local Player=game.Players.LocalPlayer

local Auto = w:CreateFolder("Auto")

--Base Variables
local Panda
local Pandas={}
local Collectibles=Workspace:FindFirstChild("Collectibles")

--Non Return Functions
local function Touch(p,t)firetouchinterest(Player.Character:WaitForChild("HumanoidRootPart"),p,0)firetouchinterest(Player.Character:WaitForChild("HumanoidRootPart"),p,1)task.wait(t)end


--Getting Data
for _, v in pairs(Collectibles:GetDescendants())do if v.Name == "Hitbox" then table.insert(Pandas, v.Parent.Name)end end
for _, v in pairs(game:GetService("ReplicatedStorage").Hidden:GetDescendants())do if v.Name == "Hitbox" then table.insert(Pandas, v.Parent.Name)end end

table.sort(Pandas, function(a, b) return a:byte() < b:byte() end)
local DropList = Auto:Dropdown("Select Panda",Pandas,true,function(SItem) --true/false, replaces the current title "Dropdown" with the option that t
    Panda = SItem
end)

Auto:Button("Teleport",function()
    spawn(function()
        for _, v in pairs(Collectibles:GetDescendants()) do
            if v.Name == Panda and v:FindFirstChild("Hitbox")then MoveTo(v.Hitbox,Vector3.new(0,0,0))end
        end
        for _, v in pairs(game:GetService("ReplicatedStorage").Hidden:GetDescendants()) do
            if v.Name == Panda and v:FindFirstChild("Hitbox")then MoveTo(v.Hitbox,Vector3.new(0,0,0))end
        end
    end)
end)

Auto:Button("Auto Find",function()
    spawn(function()
        for _, v in pairs(Collectibles:GetDescendants()) do
            if v:IsA("TouchTransmitter") then Touch(v.Parent,0.1)end
        end
        for _, v in pairs(game:GetService("ReplicatedStorage").Hidden:GetDescendants()) do
            if v:IsA("TouchTransmitter") then Touch(v.Parent,0.1)end
        end
    end)
end)
