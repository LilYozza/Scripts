local library = loadstring(game:HttpGet(('https://raw.githubusercontent.com/LilYozza/Scripts/main/Misc/UILibrary.txt')))()w = library:CreateWindow("Main")

local Auto = w:CreateFolder("Auto")
local Troll = w:CreateFolder("Troll")

local Tycoon, NilTycoon, SCollect
local Spawn = workspace:FindFirstChild("Spawn")
local AdminRoom = Spawn:FindFirstChild("Admin room")
local BlackListItems = {"Gem Teleporter"}

local function Touch(p,t)firetouchinterest(Player.Character:WaitForChild("HumanoidRootPart"),p,0)firetouchinterest(Player.Character:WaitForChild("HumanoidRootPart"),p,1)task.wait(t)end

local function GetTycoon()
    for _,v in pairs(workspace:GetDescendants()) do
        if v:IsA("StringValue") and v.Value == Player.Name then
            Tycoon = v.Parent break
        end
        if v:IsA("StringValue") and v.Value == "" and not NilTycoon and not v:FindFirstAncestor("Teleport script") then
            NilTycoon = v.Parent
        end
    end
    if Tycoon == nil then Tycoon = NilTycoon 
        for _,v in pairs(Tycoon:GetDescendants()) do
            if v:IsA("TouchTransmitter") and v:FindFirstAncestor("Become Tycoon Owner") then
                Touch(v.Parent)
                Player.character:WaitForChild("HumanoidRootPart").CFrame = v.Parent.CFrame break
            end
        end
    end
end GetTycoon()

Auto:Toggle("Auto Collect",function(bool)
    shared.toggle1 = bool
    spawn (function()
        if not Tycoon then GetTycoon() end
        while shared.toggle1 do
            for _,v in pairs(Tycoon:GetDescendants()) do
                if v:IsA("TouchTransmitter") and v.Parent.Name:match("Collect") and shared.toggle1 then
                    Touch(v.Parent)
                end
            end
        	task.wait(1)
        end
    end)
end)

Auto:Toggle("Smart Collect",function(bool)
    shared.toggle2 = bool
    SCollect = shared
end)

Auto:Toggle("Auto Buy",function(bool)
    shared.toggle3 = bool
    spawn (function()
        if not Tycoon then GetTycoon() end
        while shared.toggle3 do
            for _,v in pairs(Tycoon:FindFirstChild("Factory"):GetDescendants()) do
                if v:IsA("TouchTransmitter") and v.Parent.Parent:FindFirstChild("Humanoid") and shared.toggle3 and not v:FindFirstAncestor("OpenClose door") and not v.Parent.Parent:FindFirstChild("Humanoid").DisplayName:match("RS") then
                    price = string.gsub(v.Parent.Parent:FindFirstChild("Humanoid").DisplayName:split("-")[2], "%D", "")
                    if SCollect and tonumber(price) <= Player.leaderstats.Cash.Value + Tycoon:FindFirstChild("Cash to collect"):FindFirstChild("Money").Value then
                        Touch(Tycoon:FindFirstChild("Cash to collect"):FindFirstChild("Collect"),0.5)
                    end
                    if tonumber(price) <= Player.leaderstats.Cash.Value then
                        Touch(v.Parent,0.5) break
                    end
                    task.wait(1)
                end
            end
        	task.wait()
        end
    end)
end)

Auto:Button("Give Items",function()
    spawn(function()
        for _,v in pairs(Tycoon:FindFirstChild("Factory"):GetDescendants()) do

                Touch(v.Parent)
            end
            if v:IsA("TouchTransmitter") and v.Parent.Parent.Name:match("Giver") then
                Touch(v.Parent)
            end
        end
        for _,v in pairs(workspace:GetDescendants()) do
            if v:isA("TouchTransmitter") and v.Parent.Parent.Name:match("GearGiver") then

            end
        end

end)
