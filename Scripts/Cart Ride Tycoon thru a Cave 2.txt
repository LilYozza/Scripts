function GetTycoon()
    for _,v in pairs(workspace:GetDescendants()) do
        if v:IsA("StringValue") and v.Value == Player.Name then
            Tycoon = v.Parent break
        end
        if v:IsA("StringValue") and v.Value == "" and not NilTycoon and not v:FindFirstAncestor("Teleport script") then
            NilTycoon = v.Parent
        end
    end
    if Tycoon == nil then Tycoon = NilTycoon 
        for _,v in pairs(Tycoon:GetDescendants()) do
            if v:IsA("TouchTransmitter") and v:FindFirstAncestor("Become Tycoon Owner") then
                firetouchinterest(Player.character:WaitForChild("HumanoidRootPart"), v.Parent, 0)
                firetouchinterest(Player.character:WaitForChild("HumanoidRootPart"), v.Parent, 1)
                Player.character:WaitForChild("HumanoidRootPart").CFrame = v.Parent.CFrame break
            end
        end
    end
end
GetTycoon()

Auto:Toggle("Auto Collect",function(bool)
    shared.toggle1 = bool
    spawn (function()
        if not Tycoon then GetTycoon() end
        while shared.toggle1 do
            for _,v in pairs(Tycoon:FindFirstChild("Cash to collect"):GetDescendants()) do
                if v:IsA("TouchTransmitter") and v.Parent.Name:match("Collect") and shared.toggle1 then
                    firetouchinterest(Player.character:WaitForChild("HumanoidRootPart"), v.Parent, 0)
                    firetouchinterest(Player.character:WaitForChild("HumanoidRootPart"), v.Parent, 1) break
                end
            end
        	wait(1)
        end
    end)
end)

Auto:Toggle("Auto Buy",function(bool)
    shared.toggle3 = bool
    spawn (function()
        if not Tycoon then GetTycoon() end
        while shared.toggle3 do
            for _,v in pairs(Tycoon:FindFirstChild("Factory"):GetDescendants()) do
                if v:IsA("TouchTransmitter") and v.Parent.Parent:FindFirstChild("Humanoid") and shared.toggle3 and not v:FindFirstAncestor("OpenClose door") then
                    price = string.gsub(v.Parent.Parent:FindFirstChild("Humanoid").DisplayName:split("-")[2], "%D", "")
                    --print(price, Player.leaderstats.Cash.Value + Tycoon:FindFirstChild("Cash to collect"):FindFirstChild("Money").Value)
                    if SCollect and tonumber(price) <= Player.leaderstats.Cash.Value + Tycoon:FindFirstChild("Cash to collect"):FindFirstChild("Money").Value then
                        firetouchinterest(Player.character:WaitForChild("HumanoidRootPart"), Tycoon:FindFirstChild("Cash to collect"):FindFirstChild("Collect"), 0)
                        firetouchinterest(Player.character:WaitForChild("HumanoidRootPart"), Tycoon:FindFirstChild("Cash to collect"):FindFirstChild("Collect"), 1) wait(0.5)
                    end
                    if tonumber(price) <= Player.leaderstats.Cash.Value then
                        firetouchinterest(Player.character:WaitForChild("HumanoidRootPart"), v.Parent, 0)
                        firetouchinterest(Player.character:WaitForChild("HumanoidRootPart"), v.Parent, 1) wait(0.5)
                    end
                    wait(1)
                end
            end
        	wait()
        end
    end)
end)

Misc:Toggle("Teleport",function(bool)
    shared.Misc4 = bool
    spawn(function()
        local mouse = Player:GetMouse()
        mouse.KeyDown:Connect(function(key)
            if key == "q" and shared.Misc4 then
                if mouse.Target then
                    Player.character:WaitForChild("HumanoidRootPart").CFrame = CFrame.new(mouse.Hit.x, mouse.Hit.y + 5, mouse.Hit.z)
                end
            end
        end)
    end)
end)
